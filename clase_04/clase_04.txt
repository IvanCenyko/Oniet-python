def suma(a, b):         defino una funcion llamada suma que reciba variables n1 y n2
    resultado = a + b
    return resultado

def resta(a, b):
    resultado = suma(a, (-1)*b)         puedo operar dentro del parentesis
    return resultado

def multipli(a,b):
    repeticion = 0          contador
    valor = 0               acumulador
    while repeticion < b:   
        valor = valor + a
        repeticion = repeticion + 1
    return valor
